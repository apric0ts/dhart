139  Graph GraphGenerator::CrawlGeomParallel(UniqueQueue& todo)
140  {
141  // Generate the set of directions to use for each set of possible children
142  // Uses the maximum connection defined by the user
143  const auto directions = CreateDirecs(max_step_connection);
144 
145  // Initialize a tracker for the number of nodes that can be compared to the max nodes limit
146  int num_nodes = 0;
147 
148  RayTracer & rt_ref = this->ray_tracer;
149 
150  Graph G;
151  // Iterate through every node int the todo-list while it does not reach the maximum number of nodes limit
152  while (!todo.empty() && (num_nodes < max_nodes || max_nodes < 0))
153  {
154  // Pop nodes from the todo list (If max_nodes will be exceeded, then
155  // only pop as many as are left in max_nodes.)
156  int to_do_count = todo.size();
157  if (max_nodes > 0)
158  to_do_count = std::min(todo.size(), max_nodes - num_nodes);
159 
160  // Get as many nodes as possible out of the queue
161  auto to_be_done = todo.popMany(to_do_count);
162 
163  // If this happens, that means something is wrong with the algorithm
164  // since to_be_done with a size of zero would just cause the outer
165  // while loop to not run anymore.
166  assert(to_be_done.size() > 0);
167 
168  // Create array arrays of edges that will be used to store results
169  vector<vector<Edge>> OutEdges(to_do_count);
170 
171  // Compute valid children for every node in parallel.
172  #pragma omp parallel for schedule(dynamic) if (to_be_done.size() > 100)
173  for (int i = 0; i < to_do_count; i++)
174  {
175  // Get the parent node from the todo list at index i
176  // and cast it to a real3
177  const Node& n = to_be_done[i];
178  const auto real_parent = CastToReal3(n);
179 
180  // Generate children around this parent
181  const std::vector<real3> children = GeneratePotentialChildren(
182  real_parent,
183  directions,
184  spacing,
185  params
186  );
187 
188  // Calculate valid edges for this parent and store them in
189  // the edges array at it's index
190  OutEdges[i] = GetChildren(
191  real_parent,
192  children,
193  rt_ref,
194  params
195  );
196  }
197 
198  // Go through each edge array we just calculated, then
199  // add the edges to the graph and todo list in sequence
200  for (int i = 0; i < to_do_count; i++) {
201 
202  // Only continue if there are edges for this node
203  if (!OutEdges[i].empty()) {
204 
205  // Iterate through each edge and add it to the graph / todolist
206  for (const auto& e : OutEdges[i]) {
207  todo.push(e.child);
208  G.addEdge(to_be_done[i], e.child, e.score);
209  }
210 
211  // Increment max nodes
212  num_nodes++;
213  }
214  }
215  }
216 
217  return G;
218  }

220  Graph GraphGenerator::CrawlGeom(UniqueQueue& todo)
221  {
222  // Create directions
223  const auto directions = CreateDirecs(this->max_step_connection);
224 
225  // Cast this to a reference to avoid dereferencing every time
226  RayTracer& rt_ref = this->ray_tracer;
227 
228  int num_nodes = 0;
229  Graph G;
230  while (!todo.empty() && (num_nodes < this->max_nodes || this->max_nodes < 0)) {
231 
232  // Get the parent node from the todo list
233  const auto parent = todo.pop();
234 
235  // To maintain our precision standards, we'll convert this node to a real3.
236  const auto real_parent = CastToReal3(parent);
237 
238  // Generate children from parent
239  const std::vector<real3> children = GeneratePotentialChildren(
240  real_parent,
241  directions,
242  spacing,
243  params
244  );
245 
246  // Create edges
247  std::vector<graph_edge> edges = GetChildren(
248  real_parent,
249  children,
250  rt_ref,
251  params
252  );
253 
254  // Add new nodes to the queue. It'll drop them if they
255  // already were evaluated, or already existed on the queue
256  for (auto edge : edges)
257  todo.PushAny(edge.child);
258 
259  // Add new edges to the graph
260  if (!edges.empty())
261  for (const auto& edge : edges)
262  G.addEdge(parent, edge.child, edge.score);
263 
264  // Increment node count
265  ++num_nodes;
266  }
267  return G;
