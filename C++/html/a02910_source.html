714  // 0 1 2
715 
716  // indices[0] refers to vertices[0] (beginning of vertex 0)
717  // indices[1] refers to vertices[3] (beginning of vertex 1)
718  // indices[2] refers to vertices[6] (beginning of vertex 2)
719  std::vector<int> indices_0 = { 0, 1, 2 };
720  std::vector<int> indices_1 = { 0, 1, 2 };

725  bool equivalent = mesh_0 == mesh_1; // returns true
726 
727  // operator== will determine if two MeshInfo are equal if
728  // the Euclidean distance between each matching element in mesh_0 and mesh_1
729  // is within 0.001. This means that for all i between vertices_0 and vertices_1 --
730  // each x, y, z within
731  // vertices_0[i] and vertices_1[i]
732  // must be within 0.001 to be considered equivalent.
733 
734  // Of course, this also means that if
735  // mesh_0.NumVerts() != mesh_1.NumVerts(),
736  // mesh_0 and mesh_1 are not equivalent at all.
