95 // A quick copy of the method to construct a path from a distnace
96 // and predecessor matrix in the pathfinder with some modifications.
97 // Using this to test distance and predecessor matricies.
98 inline Path QuickConstructPath(
99  int start,
100  int end,
101  const std::vector<int> & pred,
102  const std::vector<float> & distances
103 ) {
104 
105  // Create a new path and add the end point.
106  Path p;
107  p.AddNode(end, 0);
108 
109  int current_node = end;
110  if (pred[current_node] == current_node) return Path{};
111 
112  float last_cost = distances[current_node];
113 
114  while (current_node != start) {
115  int next_node = pred[current_node];
116  float current_cost = distances[next_node];
117  float de_facto_cost = last_cost - current_cost;
118 
119  p.AddNode(next_node, de_facto_cost);
120 
121  last_cost = current_cost;
122  current_node = next_node;
123  }
124 
125  // Flip the order of this since this algorithm generates it from end to start
126  p.Reverse();
127  return p;
128 }
