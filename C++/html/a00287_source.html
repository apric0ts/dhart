39  void CheckForDuplicates(const HF::SpatialStructures::Graph& g, float precision = 0.0001f) {
40 
41  // Get nodes from the graph
42  const auto nodes = g.Nodes();
43 
44  // Setup variables
45  int duplicate_pairs = 0;
46  int duplicates_this_node = 0;
47  int duplicate_nodes = 0;
48 
49  // Iterate through every node in the graph
50  for (const auto& node : nodes) {
51  --duplicate_pairs; // subrtact one because this node will be compared to itself atleast once
52  duplicates_this_node = 0;
53 
54  // Iterate through every other node in the graph
55  for (const auto& node2 : nodes)
56  {
57  // Check if this node closer to node2 than our precision value
58  if (node.distanceTo(node2) < precision) {
59 
60  // If it is, increment duplciate pairs
61  ++duplicate_pairs;
62  ++duplicates_this_node;
63 
64  // If this has had 2 duplicates (itself and atleast one other node) then it is a duplicate and duplicates
65  // should be incremented
66  if (duplicates_this_node == 2)
67  duplicate_nodes++;
68  }
69  }
70  }
71 
72 
73  // Divide duplicate nodes by 2, since both duplicates would increment this counter
74  // and technically only one is the original
75  duplicate_nodes /= 2;
76  int total_pairs = g.size() * g.size() - 1;
77  float duplicate_node_percentage = (static_cast<float>(duplicate_nodes) / static_cast<float>(g.size())) * 100.00f;
78  float duplicate_pair_percentage = (static_cast<float>(duplicate_pairs) / static_cast<float>(g.size() * g.size())) * 100.00f;
79 
80  // print to console
81  std::cerr << duplicate_nodes << "/" << g.size() << "(" << duplicate_node_percentage << "\%) Duplicate Nodes" << std::endl;
82  std::cerr << duplicate_pairs << "/" << total_pairs << "(" << duplicate_pair_percentage << "\%) Duplicate Pairs" << std::endl;
83 
84  // Fail if there are duplicate pairs
85  EXPECT_EQ(0, duplicate_pairs);
86  }
