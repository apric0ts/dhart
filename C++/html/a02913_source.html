247  double RayTriangleIntersection(
248  const Vector3D& origin,
249  const Vector3D& direction,
250  const Vector3D& v1,
251  const Vector3D& v2,
252  const Vector3D& v3)
253  {
254 
255  const double EPSILON = 0.0000001;
256  const Vector3D inverted_direction = direction;//InvertVector(direction);
257 
258  const Vector3D edge1 = v2 - v1;
259  const Vector3D edge2 = v3 - v1;
260  const Vector3D h = cross(inverted_direction, edge2);
261  const double a = dot(edge1, h);
262 
263  // This ray is parallel to this triangle.
264  if (a > -EPSILON && a < EPSILON)
265  return -1;
266 
267  const double f = 1.0 / a;
268  const Vector3D s = origin - v1;
269  const double u = f * dot(s, h);
270  if (u < 0.0 || u > 1.0)
271  return -1;
272 
273  const Vector3D q = cross(s, edge1);
274  const double v = f * dot(direction, q);
275  if (v < 0.0 || u + v > 1.0)
276  return -1;
277 
278  // At this stage we can compute t to find out where the intersection point is on the line.
279  double t_hit = f * dot(edge2, q);
280  return t_hit;
281  }
