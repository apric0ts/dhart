193  inline void Aggregate(float& out_total, float new_value, const AGGREGATE_TYPE agg_type, int count = 0) {
194  switch (agg_type) {
195  case AGGREGATE_TYPE::COUNT:
196  if (new_value > 0) out_total += 1;
197  break;
198  case AGGREGATE_TYPE::SUM:
199  out_total += new_value;
200  break;
201  case AGGREGATE_TYPE::AVERAGE: {
202  int n = count - 1;
203  out_total = (n * (out_total)+new_value) / count;
204  break;
205  }
206  case AGGREGATE_TYPE::MAX:
207  out_total = std::max<float>(out_total, new_value);
208  break;
209  case AGGREGATE_TYPE::MIN:
210  out_total = std::min<float>(out_total, new_value);
211  break;
212  default:
213  throw std::out_of_range("Unimplemented aggregation type");
214  break;
215  }
216  assert(out_total == 0 || isnormal(out_total));
217  return;
218  }
