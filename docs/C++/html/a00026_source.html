198  inline void Aggregate(float& out_total, float new_value, const AGGREGATE_TYPE agg_type, int count = 0) {
199  switch (agg_type) {
200  case AGGREGATE_TYPE::COUNT:
201  if (new_value > 0) out_total += 1;
202  break;
203  case AGGREGATE_TYPE::SUM:
204  out_total += new_value;
205  break;
206  case AGGREGATE_TYPE::AVERAGE: {
207  int n = count - 1;
208  out_total = (n * (out_total)+new_value) / count;
209  break;
210  }
211  case AGGREGATE_TYPE::MAX:
212  out_total = std::max<float>(out_total, new_value);
213  break;
214  case AGGREGATE_TYPE::MIN:
215  out_total = std::min<float>(out_total, new_value);
216  break;
217  default:
218  throw std::out_of_range("Unimplemented aggregation type");
219  break;
220  }
221  assert(out_total == 0 || isnormal(out_total));
222  return;
223  }
