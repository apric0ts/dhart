198  void MeshInfo::PerformRotation(float rx, float ry, float rz)
199  {
200  // Convert to radians
201  float radian_ratio = static_cast<float>(M_PI) / 180.00f;
202  rx *= radian_ratio; ry *= radian_ratio; rz *= radian_ratio;
203 
204  // Implementation based on
205  // https://stackoverflow.com/questions/21412169/creating-a-rotation-matrix-with-pitch-yaw-roll-using-eigen
206  Eigen::AngleAxis<float> rollAngle(rz, Eigen::Vector3f::UnitZ());
207  Eigen::AngleAxis<float> yawAngle(ry, Eigen::Vector3f::UnitY());
208  Eigen::AngleAxis<float> pitchAngle(rx, Eigen::Vector3f::UnitX());
209 
210 
211  // Create a quaternion from the angles, normalize it, then
212  // convert it to a rotation matrix.
213  Eigen::Quaternion<float> q = (rollAngle * yawAngle * pitchAngle);
214  q.normalize();
215  Eigen::Matrix3f rotation_matrix = q.toRotationMatrix();
216 
217  // Assert that we didn't create any NANS or infinite values
218  assert(rotation_matrix.allFinite());
219 
220  // Apply the rotation matrix to verts
221  verts = (rotation_matrix * verts);
222 
223  // Once again assert that we didn't create any nans or infinite numbers.
224  assert(verts.allFinite());
225  }
