43  std::set<std::pair<int, int>> permutations(int limit) {
44  // Create a vector of all numbers between 1 and limit + 1, as well as their inverses
45  vector<int> steps;
46  steps.reserve(2 * limit);
47 
48  for (int i = 1; i < limit + 1; i++) {
49  steps.push_back(i);
50  steps.push_back(-i);
51  }
52 
53  std::set<std::pair<int, int>> perms;
54 
55  // Nest a for loop to capture every permutation
56  for (int j : steps)
57  for (int k : steps)
58  if (abs(j) != abs(k))
59  perms.emplace(pair(j, k));
60 
61  return perms;
62  }

178  bool GraphGeneratorPrivate::isUpSlope(const Node& n1, const Node& n2)
179  {
180  // Slope is rise/run
181  const float run = sqrtf(powf((n1.x - n2.x), 2) + powf((n1.y - n2.y), 2));
182  const float rise = n2.z - n1.z;
183 
184  // Calculate the angle between rise and run.
185  const float calc_slope = atan2(rise, run) * (180.00f / static_cast<float>(M_PI));
186 
187  // Check against downslope and upslope.
188  return calc_slope > -1 * GG.downslope && calc_slope < GG.upslope;
189  }

228  STEP GraphGeneratorPrivate::CheckConnection(const Node& parent, const Node& child)
229  {
230  // Create a copy of parent and child that can be modified
231  auto node1 = parent;
232  auto node2 = child;
233 
234  // Offset them from the ground slightly
235  node1[2] += 0.001f;
236  node2[2] += 0.001f;
237 
238  // See if there's a direct line of sight between parent and child
239  if (!OcclusionCheck(node1, node2)) {
240 
241  // If there is a direct line of sight, and they're on the same plane
242  // then there is no step.
243  if (abs(node1.z - node2.z) < 0.001f) return STEP::NONE;
244 
245  // Check if we can traverse the slope from parent to child.
246  else if (isUpSlope(parent, child)) return STEP::NONE;
247 
248  return STEP::NOT_CONNECTED;
249  }
250 
251  // Otherwise check for a step based connectiom
252  else {
253 
254  STEP s = STEP::NONE;
255 
256  // If parent is higher than child, then offset child downwards
257  if (node1[2] > node2[2]) {
258  node1[2] = node1[2] + GG.downstep;
259  node2[2] = node2[2] + 0.001f;
260  s = STEP::DOWN;
261  }
262 
263  // If parent is lower than child, offset child upwards by upstep
264  else if (node1[2] < node2[2]) {
265  node1[2] = node1[2] + GG.upstep;
266  node2[2] = node2[2] + 0.001f;
267  s = STEP::UP;
268  }
269 
270  // If they're on an equal plane then offset by upstep to see
271  // if the obstacle can be stepped over.
272  else if (node1[2] == node2[2]) {
273  node1[2] = node1[2] + GG.upstep;
274  node2[2] = node2[2] + 0.001f;
275  s = STEP::OVER;
276  }
277 
278  // If there is a line of sight then the nodes are connected
279  // with the step type we calculated
280  if (!OcclusionCheck(node1, node2)) {
281  return s;
282  }
283 
284  // If not, then there is no connection between these
285  // nodes.
286  return STEP::NOT_CONNECTED;
287  }

316  void GraphGeneratorPrivate::CrawlGeomParallel(UniqueQueue & todo)
317  {
318  auto directions = CreateDirecs(GG.max_step_connection);
319 
320  // Create a vector to use for edges
321  vector<Edge> links;
322  int num_nodes = 0;
323 
324  // Iterate through every node int the todo-list
325  while (!todo.empty() && (num_nodes < GG.max_nodes || GG.max_nodes < 0)) {
326 
327  // If the todo list is big enough, go into parallel;
328  if (todo.size() >= 100) {
329

331  // only pop as many as are left in max_nodes.)
332  int to_do_count = todo.size();
333  if (GG.max_nodes > 0)
334  to_do_count = std::min(todo.size(), GG.max_nodes - num_nodes);
335 
336  // Get every node out of the queue
337  auto to_be_done = todo.popMany(to_do_count);
338 
339  // If this happens, that means something is wrong with the algorithm
340  // since to_be_done with a size of zero would just cause the outer
341  // while loop to not run anymore.
342  assert(to_be_done.size() > 0);
343

346

348  #pragma omp parallel
349  {
350  #pragma omp for schedule(dynamic)
351  for (int i = 0; i < to_do_count; i++) {
352  const Node n = to_be_done[i];

354  }
355  }
356 
357  // Now out of parallel we add every node to to the todo list in order
358  // and add the parent to the graph
359  for (int i = 0; i < to_do_count; i++) {

361

363  // The queue will automatically discard nodes that have already
364  // been in it before.
365  for (auto& e : OutEdges[i]) {
366  todo.push(e.child);

368  }
369

371  num_nodes++;
372  }
373  }
374  }
375 
376  // Otherwise go in sequence (identical to CrawlGeom)
377  else {
378  const Node parent = todo.pop();
379  links.clear();
380 
381  ComputerParent(parent, directions, links);
382 
383  for (auto edge : links)
384  todo.push(edge.child);
