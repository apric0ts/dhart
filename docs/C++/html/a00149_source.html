130  inline void Aggregate(float& out_total, float new_value, const COST_AGGREGATE agg_type, int & count)
131  {
132  switch (agg_type) {
133  case COST_AGGREGATE::COUNT:
134  if (new_value > 0) count++;
135  out_total = count;
136  break;
137  case COST_AGGREGATE::SUM:
138  out_total += new_value;
139  break;
140  case COST_AGGREGATE::AVERAGE: {
141  // Ensure the number of elements isn't zero.
142  int n = std::max(count - 1, 1);
143 
144  out_total = (n * (out_total)+new_value) / std::max(count, 1);
145  count++;
146  break;
147  }
148  default:
149  throw std::out_of_range("Unimplemented aggregation type");
150  break;
151  }
152  assert((out_total == 0 || isnormal(out_total)));
153  return;
154  }

156  std::vector<float> Graph::AggregateGraph(COST_AGGREGATE agg_type, bool directed) const
157  {
158  // This won't work if the graph isn't compressed.
159  if (this->needs_compression) throw std::exception("The graph must be compressed!");
160 
161  // Create an array of floats filled with zeros.
162  vector<float> out_costs(this->size(), 0);
163 
164  // If directed is true, then we only need the values in a node's row to calculate it's score.
165  if (directed)
166  for (int k = 0; k < edge_matrix.outerSize(); ++k) {
167 
168  // Sum all values in the row for node k
169  float sum = edge_matrix.row(k).sum();
170 
171  // Get the count of non_zeros for node k
172  int count = edge_matrix.row(k).nonZeros();
173 
174  // Run aggregate once with the sum and count of the non-zeros in row k
175  Aggregate(out_costs[k], sum, agg_type, count);
176 
177  }
178 
179  // Based on implementation from Eigen's sparse matrix tutorial: Iterating over the nonzero
180  // coefficents: https://eigen.tuxfamily.org/dox/group__TutorialSparse.html If directed is
181  // false, then every edge needs to be iterated through since edges that go to a node also
182  // count towards its score.
183  else {
184  // We need to keep track of count per node as well for this algorithm
185  vector<int> count(this->size(), 0);
186 
187  // Iterate through every node in the graph
188  for (int k = 0; k < edge_matrix.outerSize(); ++k) {
189 
190  // Iterate through every edge for the node at column k
191  for (SparseMatrix<float, 1>::InnerIterator it(edge_matrix, k); it; ++it)
192  {
193  // Get values from the iterator for this row/col.
194  float cost = it.value();
195  int child = it.col();
196  int parent = it.row();
197 
198  // Aggregate costs for both the parent and the child.
199  Aggregate(out_costs[parent], cost, agg_type, count[parent]);
200  Aggregate(out_costs[child], cost, agg_type, count[child]);
201  }
202  }
203  }
204  return out_costs;
