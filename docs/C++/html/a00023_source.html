119  int num_points,

123 
124  // Convert limits to radians
125  const float upperlimit = ConvertToRadians(upwards_fov);
126  const float lowerlimit = ConvertToRadians(downward_fov);
127 
128  // Calculate maximum and minimum altitude
129  const float max_phi = static_cast<float>(M_PI_2) + lowerlimit;
130  const float min_phi = static_cast<float>(M_PI_2) - upperlimit;
131 
132  const int n = num_points;
133  std::vector<std::array<float, 3>> out_points;
134  float offset = 2.0f / float(n);
135  const float increment = static_cast<float>(M_PI)* (3.0f - sqrtf(5.0f));
136 
137  for (int i = 5; i < n + 5; i++) {
138  float y = ((float(i) * offset) - 1.0f) - (offset / 0.2f);
139  float r = sqrtf(1.0f - powf(y, 2));
140 
141  float phi = (float(i + 1) * increment);
142  float x = cosf(phi) * r;
143  float z = sinf(phi) * r;
144 
145  // Check for nans. These can sometimes occur at the beginning and end of the result set
146  // due to imprecision.
147  if (!isnan(x) && !isnan(y) && !isnan(z)) {
148  auto new_point = std::array<float, 3>{x, y, z};
149  Normalize(new_point);
150  if (AltitudeWithinRange(new_point, max_phi, min_phi))
151  out_points.emplace_back(new_point);
152  }
153  }
