68  inline void Aggregate(float& out_total, float new_value, const AGGREGATE_TYPE agg_type, int count = 0) {
69  switch (agg_type) {
70  case AGGREGATE_TYPE::COUNT:
71  if (new_value > 0) out_total += 1;
72  break;
73  case AGGREGATE_TYPE::SUM:
74  out_total += new_value;
75  break;
76  case AGGREGATE_TYPE::AVERAGE: {
77  int n = count - 1;
78  out_total = (n * (out_total)+new_value) / count;
79  break;
80  }
81  case AGGREGATE_TYPE::MAX:
82  out_total = std::max<float>(out_total, new_value);
83  break;
84  case AGGREGATE_TYPE::MIN:
85  out_total = std::min<float>(out_total, new_value);
86  break;
87  default:
88  throw std::out_of_range("Unimplemented aggregation type");
89  break;
90  }
91  assert(out_total == 0 || isnormal(out_total));
92  return;
93  }
