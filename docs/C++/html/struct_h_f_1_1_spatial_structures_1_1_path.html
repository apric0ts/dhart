<< "Node: " << curr->node << std::endl;
++curr;
}

Determines if all PathMember in p1 and p2's underlying PathMember vector are equivalent, or not, see operator== for PathMember

Parameters
p2The Path to compare against p1 (or rather, the elements of their PathMember vectors)
Returns
True, if all PathMember elements in p2 are equal to all PathMember elements in p1, false otherwise
PathMember p0 = { 3.14f, 3 }; // Create all the PathMember objects
PathMember p1 = { 2.78f, 2 };
PathMember p2 = { 3.14f, 3 };
PathMember p3 = { 2.78f, 2 };
PathMember p4 = { 1.1f, 9 };
PathMember p5 = { 123.0f, 10 };
std::vector<PathMember> members_0{p0, p1}; // Create the PathMember vectors
std::vector<PathMember> members_1{p2, p3};
std::vector<PathMember> members_2{p4, p5};
std::vector<PathMember> members_3{p1, p0};
Path path_0(members_0); // Create the Path objects
Path path_1(members_1);
Path path_2(members_2);
Path path_3(members_3);
bool same_values_same_order = path_0 == path_1;
bool totally_different = path_0 == path_2;
bool same_values_different_order = path_0 == path_3;
// path_0 and path_1 share the same PathMember values, with an identical permutation,
// so they are equivalent.
// path_0 and path_2 are not equivalent, because they have completely different PathMember
// values.
// path_0 and path_3 are not equivalent, because although they have PathMember objects of
// the same values, the order in which path_0 and path_3 have their member vectors arranged
// are different.
