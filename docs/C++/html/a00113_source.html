77  inline Path ConstructShortestPathFromPred(
78  int start,
79  int end,
80  const std::vector<size_t>& pred,
81  const std::vector<float>& distances
82  ) {
83 
84  // Create a new path and add the end point.
85  Path p;
86  p.AddNode(end, 0);
87 
88  // Return an empty path if there's no path from start to end indicated by the predecessor's
89  // value for the current node being the current node.
90  int current_node = end;
91  if (pred[current_node] == current_node) return Path{};
92 
93  float last_cost = distances[current_node];
94 
95  // Follow the predecessor matrix from end to start
96  while (current_node != start) {
97 
98  // If this is triggered, there's something wrong with this algorithm because the path
99  // suddenly has more nodes than there are in the entire graph.
100  if (p.size() > pred.size())
101  throw std::exception("Path included more nodes than contaiend in the graph!");
102 
103  // Get the next node from the predecessor matrix
104  int next_node = pred[current_node];
105 
106  // Get the current cost for traversing the graph so far.
107  float current_cost = distances[next_node];
108 
109  // Subtract the last cost from the graph to get the cost of traversing this node
110  float de_facto_cost = last_cost - current_cost;
111 
112  // Add this node to the list
113  p.AddNode(next_node, de_facto_cost);
114 
115  // Save this node and cost as the last node/cost
116  last_cost = current_cost;
117  current_node = next_node;
118  }
119 
120  // Flip the order of this since this algorithm generates it from end to start
121  p.Reverse();
122  return p;
123  }
