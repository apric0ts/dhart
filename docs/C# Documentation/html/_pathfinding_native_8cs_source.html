50 
51  // Find the path in unmanaged code. If this returns OK, then the parameters
52  // of out_ptrs will be updated with the data for the path.
53  HF_STATUS res = CreatePath(
54  graph_ptr,
55  start,
56  end,

58  ref out_ptrs.size,
59  ref out_ptrs.vector,

61  );
62 
63  // If NO_PATH is returned, that means no path exists from start to end
64  // so make sure our output reflects this.
65  if (res == HF_STATUS.OK)
66  return out_ptrs;
67  else if (res == HF_STATUS.NO_PATH)
68  {
69  // If NO_PATH, that just means that there is on path between start and end,
70  // so just return an invalid CVectorAndData **to signal that this isn't possble
71  out_ptrs.size = -1;
72  out_ptrs.data = IntPtr.Zero;
73  out_ptrs.vector = IntPtr.Zero;
74  }
75  // Throw if the cost doesn't exist.
76  else if (res == HF_STATUS.NO_COST) // Cost type doesn't exist in the graph
77  throw new KeyNotFoundException("cost_type " + cost_type + " is not the valid key of a cost in the graph");
78  else
79  // If this ever fails, something changed in the CInterface that wasn't reflected in C#
80  Debug.Assert(false);
81  return out_ptrs;
