174 internal static void C_AddEdge(IntPtr graph_ptr, Vector3D parent, Vector3D child, float cost, string cost_type)
175 {
176 // Convert parent and child to arrays
177 var parent_arr = parent.ToArray();
178 var child_arr = child.ToArray();
179
180 // Try to add them to the graph.
181 HF_STATUS stat = AddEdgeFromNodes(graph_ptr, parent_arr, child_arr, cost, cost_type);
182
183 // Handle error codes
184 switch (stat)
185 {
186 // If it's ok then no error checking required
187 case HF_STATUS.OK:
188 return;
189
190 // if NOT_COMPRESSED then inform the user

195 case HF_STATUS.OUT_OF_RANGE:
196 throw new InvalidCostOperation("Tried to add an alternate cost to an edge that doesn't already" +

334 internal static CVectorAndData C_AggregateEdgeCosts(
335 IntPtr graph_ptr,
336 bool directed,
337 GraphEdgeAggregation agg_type,
338 string cost_type
339 )
340 {
341 // Create a new CVector and data to return
342 CVectorAndData out_ptrs = new CVectorAndData();
343
344 // Aggregate costs in C++, updating out_ptrs and out_ptrs.data
345 var res = AggregateCosts(graph_ptr, agg_type, directed, cost_type, ref out_ptrs.vector, ref out_ptrs.data);
346
347 Debug.Assert(res != HF_STATUS.GENERIC_ERROR); // An exception that wasn't given an error code was caught by C++
348 Debug.Assert(res != HF_STATUS.NOT_COMPRESSED); // The graph wasn't compressed before calling this function.
349 // Ensure callers aren't allowed to make this happen.
350 // Handle return types
351 switch (res)
352 {
353 // If ok then no further action is needed
354 case HF_STATUS.OK:
355 break;
356 // Inform the user if the graph isn't compressed
357 case HF_STATUS.NOT_COMPRESSED:
358 throw new LogicError("The graph wasn't compressed.");
359 // If the specified cost type doesn't exist, then inform the user
360 case HF_STATUS.NO_COST:
361 throw new KeyNotFoundException("The cost type '" + cost_type + "' did not exist already in the graph");
362 default:
363 Debug.Assert(false); // There's an error code being returned that
364 // isn't accounted for. All outputs from C++ should be handled
365 // by explicit asserts or new exceptions.
366 break;
367 }
368
369 // Return out_ptrs
370 return out_ptrs;
