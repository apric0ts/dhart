46 internal static IntPtr[] C_LoadOBJ(string obj_path, float xrot, float yrot, float zrot, GROUP_METHOD gm)
47 {
48 // Create a pointer as an output parameter
49 IntPtr out_ptr = new IntPtr();
50
51 // Call the function in C++ and capture the error code
52 int num_meshes = 0;
53 HF_STATUS res = LoadOBJ(obj_path, (int)gm, xrot, yrot, zrot, ref out_ptr, ref num_meshes);
54
55 // If the file was not found, throw the standard file not found exception
56 if (res == HF_STATUS.NOT_FOUND)
57 throw new FileNotFoundException(obj_path + " did not lead to any file");
58
59 // If the file at the filepath didn't represent a valid OBJ model, throw InvalidMesh Exception
60 else if (res == HF_STATUS.INVALID_MESH)
61 throw new InvalidMeshException(obj_path + " did not lead to an obj file, or the obj file was invalid!");
62
63 // Iterate through the returned pointer array, and marshall each of it's pointers
64 // into managed memory
65 IntPtr[] out_ptrs = new IntPtr[num_meshes];
66 //for (int i = 0; i < num_meshes; i++)
67 // out_ptrs[i] = Marshal.ReadIntPtr(out_ptr, i * sizeof(IntPtr));
68
69 Marshal.Copy(out_ptr, out_ptrs, 0, num_meshes);
70
71 // Free the memory allocated by C++ for the pointer array
72 DestroyMeshInfoPtrArray(out_ptr);
73
74 // Return managed pointer array
75 return out_ptrs;
76 }

131 internal static IntPtr C_StoreMesh(IEnumerable<float> vertices, IEnumerable<int> indices, string name, int id)
132 {
133
134 // Convert IEnumerables to arrays
