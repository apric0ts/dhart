32 internal static IntPtr C_AllToAllVisibilityGraph(IntPtr bvh, IEnumerable<Vector3D> nodes, float height, bool directed = false)
33 {
34 // Setup out_ptr to be updated when we call this function
35 IntPtr out_ptr = new IntPtr();
36
37 // Flatten array of nodes
38 var flat_nodes = HelperFunctions.FlattenVectorArray(nodes);
39
40 // Get Generate the visibility graph in C++, using the undirected function if it's undirected
41 // or using the direceted function if it's directed
42 HF_STATUS res;
43 if (directed == true)
44 res = CreateVisibilityGraphAllToAll(bvh, flat_nodes, nodes.Count(), ref out_ptr, height);

46 res = CreateVisibilityGraphAllToAllUndirected(bvh, flat_nodes, nodes.Count(), ref out_ptr, height);
47
48 // If no graph could be generated, return null
