183 public static Path[] DijkstraShortestPathMulti(
184 Graph graph,
185 int[] start_ids,
186 int[] end_ids,
187 string cost_type = ""
188 )
189 {
190 // If our precondition fails, throw
191 if (start_ids.Length != end_ids.Length)
192 throw new ArgumentException("Length of start_ids didn't equal length of end_ids");
193
194 // Call the native function to generate cvectors and data
195 CVectorAndData[] cvads = NativeMethods.C_CreatePaths(graph.Pointer, start_ids, end_ids, cost_type);
196
197 // Create an array of paths to hold all paths created by the call to native code.
198 Path[] paths = new Path[start_ids.Length];
199
200 // Iterate through each CVectorAndData in the returned array
201 int size = start_ids.Length;
202 for (int i = 0; i < size; i++)
203 {
204
205 // If this path is valid, then jsut
206 if (cvads[i].IsValid())
207 paths[i] = new Path(cvads[i]);
208
209 // If this CVectorAndData isn't valid that indicates a path
210 // couldn't be created between the start and end node at this
211 // position, insert a null element at this index.
212 else
213 paths[i] = null;
214 }
215
216 return paths;
217 }

259 public static Path[] DijkstraShortestPathMulti(
260 Graph graph,
261 IEnumerable<Vector3D> start_nodes,
262 IEnumerable<Vector3D> end_nodes,
263 string cost_type = ""
264 )
265 {
266 // If the size of start and end nodes differ, then return false
267 if (start_nodes.Count() != end_nodes.Count())
268 throw new ArgumentException("Length of start_nodes didn't equal length of end_nodes");
269
270 // Create ID arrays
271 int size = start_nodes.Count();
272 int[] start_ids = new int[size];
273 int[] end_ids = new int[size];
274
275 // Get the IDS of every start and end node using a foreach loop here.
276 // Note: that foreach is slightly faster than int loops due to .net optimizations. This
277 // may not matter much for this function, but it's an example
278 // of how some procedures can be sped up.
279
286 int i = 0;
287 foreach (var start_end in start_nodes.Zip(end_nodes, (start, end) => new Tuple<Vector3D, Vector3D>(start, end)))
288 {
289 // Insert these elements into start_ids and end_ids at index i
290 start_ids[i] = graph.GetNodeID(start_end.Item1);
291 end_ids[i] = graph.GetNodeID(start_end.Item2);
292 i++;
293 }
294
295 // Return the results of the ID overload.
296 return DijkstraShortestPathMulti(graph, start_ids, end_ids, cost_type);
297 }
