106 internal static void MarshalUnmananagedArray2Struct<T>(IntPtr unmanagedArray, int length, out T[] mangagedArray) where T : struct
107 {
108 // Calculate the size T in bytes
109 var size = Marshal.SizeOf(typeof(T));
110
111 // Construct a new managed array of T that can fit all elements of UnmanagedArray
112 mangagedArray = new T[length];
113
114 // Iterate through every element in UnmanagedArray
115 for (int i = 0; i < length; i++)
116 {
117 // Calculate the position of the element in unnmanagedArray at index i
118 IntPtr ins = new IntPtr(unmanagedArray.ToInt64() + i * size);
119
120 // Marshal this pointer to a struct, then insert it into the same index in
121 // the managed array
122 mangagedArray[i] = (T)Marshal.PtrToStructure(ins, typeof(T));
123 }
124 }

210 public static Vector3D[] FloatArrayToVectorArray(float[] float_array, bool[] result_array)
211 {
212 // If our precondition isn't met, throw
213 int size = float_array.Length / 3;
214 if (size != result_array.Length)
215 throw new ArgumentException("Size of float_array was not equal to the size of the result array!");
216
217 Vector3D[] out_points = new Vector3D[size];
218
219 // Iterate through every result
220 for (int i = 0; i < size; i++)
221 {
222
223 // Calculate the index of this point in float_array
224 int os = i * 3;
225
226 // If the result array indicated an intersection here, insert a new vector
227 // at this point containing the x,y, and z coordinates of the new point
228 if (result_array[i])
229 {
230 float x = float_array[os];
231 float y = float_array[os + 1];
232 float z = float_array[os + 2];
233 out_points[i] = new Vector3D(x, y, z);
234 }
235 else // Return an invalid point to signify that no geometry was intersected
