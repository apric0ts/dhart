196 internal static Vector3D[] C_IntersectPoints(
197 IntPtr ert,
198 float[] origins,
199 float[] directions,
200 float max_distance
201)
202 {
203 // Get the size of both arrays
204 int num_origins = origins.Length / 3;
205 int num_directions = directions.Length / 3;
206
207 // Declare result here, since it needs to be written to later
208 HF_STATUS res = HF_STATUS.GENERIC_ERROR;
209 Vector3D[] out_points;
210
211 // Depending on the configuration, this will call different functions in the
212 // C-Interface. All accomplish the same thing: Call a function, interpret the results,
213 // capture the return code.
214 if (num_origins == num_directions)
215 {
216 bool[] result_array = new bool[num_origins];
217 res = CastMultipleRays(ert, origins, directions, num_origins, max_distance, result_array);
218 out_points = HelperFunctions.FloatArrayToVectorArray(origins, result_array);
219 }
220 else if (num_origins > num_directions && num_directions == 1)
221 {
222 bool[] result_array = new bool[num_origins];
223 res = CastMultipleOriginsOneDirection(ert, origins, directions, num_origins, max_distance, result_array);
224 out_points = HelperFunctions.FloatArrayToVectorArray(origins, result_array);
225 }
226 else if (num_directions > num_origins && num_origins == 1)
227 {
228 bool[] result_array = new bool[num_directions];
229 res = CastMultipleDirectionsOneOrigin(ert, origins, directions, num_directions, max_distance, result_array);
230 out_points = HelperFunctions.FloatArrayToVectorArray(directions, result_array);
231 }
232 else // If it doesn't match any configuration, throw
233 throw new ArgumentException("Arguments did not match one of the predefined cases!");
234
235 // If this is ever not HF_STATUS.OK, then something changed in the C-Interface that wasn't reflected here.
236 if (res != HF_STATUS.OK) Debug.Assert(false);
237
238 return out_points;
239 }
