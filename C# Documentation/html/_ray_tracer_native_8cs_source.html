176  float[] flat_dirs = NativeUtils.HelperFunctions.FlattenVectorArray(directions);
177 
178  // Select C function to use based on number of origins/directions
179  int num_origins = origins.Count();
180  int num_directions = directions.Count();
181 
182  // Declare result here, since it needs to be written to later
183  HF_STATUS res = HF_STATUS.GENERIC_ERROR;
184  Vector3D[] out_points;
185 
186  // Depending on the configuration, this will call different functions in the
187  // C-Interface. All accomplish the same thing: Call a function, interpret the results,
188  // capture the return code.
189  if (num_origins == num_directions)
190  {
191  bool[] result_array = new bool[num_origins];
192  res = CastMultipleRays(ert, flat_origins, flat_dirs, num_origins, max_distance, result_array);
193  out_points = HelperFunctions.FloatArrayToVectorArray(flat_origins, result_array);
194  }
195  else if (num_origins > num_directions && num_directions == 1)
196  {
197  bool[] result_array = new bool[num_origins];
198  res = CastMultipleOriginsOneDirection(ert, flat_origins, flat_dirs, num_origins, max_distance, result_array);
199  out_points = HelperFunctions.FloatArrayToVectorArray(flat_origins, result_array);
200  }
201  else if (num_directions > num_origins && num_origins == 1)
202  {
203  bool[] result_array = new bool[num_directions];
204  res = CastMultipleDirectionsOneOrigin(ert, flat_origins, flat_dirs, num_directions, max_distance, result_array);
205  out_points = HelperFunctions.FloatArrayToVectorArray(flat_dirs, result_array);
206  }
207  else // If it doesn't match any configuration, throw
208  throw new ArgumentException("Arguments did not match one of the predefined cases!");
209 
210  // If this is ever not HF_STATUS.OK, then something changed in the C-Interface that wasn't reflected here.
211  if (res != HF_STATUS.OK) Debug.Assert(false);
212 
213  return out_points;
